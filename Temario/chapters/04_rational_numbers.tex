\chapter{Números racionales.}
\section{El cuerpo ordenado de los números racionales}

Sobre $\mathbb{Z}\times\pr{\mathbb{Z}\setminus\set{0}}$, definimos la relación $\thicksim$ dada por $\pr{a,b}\thicksim\pr{c,d}$ si y solamente si $a\cdot d=b\cdot c$. Definimos el conjunto de los números racionales como sigue:

\begin{equation}
    \mathbb{Q}:=\frac{\mathbb{Z}\times\pr{\mathbb{Z}\setminus\set{0}}}{\thicksim}.
\end{equation}

Dados $a,b\in\mathbb{Z}$ con $b\neq 0$, la clase de equivalencia $\br{\pr{a,b}}$ la denotaremos por $\displaystyle\frac{a}{b}$. 

\begin{equation}
    \mathbb{Q}=\set{\frac{a}{b}\middle|a,b\in\mathbb{Z},b\neq 0}.
\end{equation}

A los representantes de cada número racional los llamaremos fracciones. Diremos que la primera componente de una fracción es el numerador y la segunda el denominador.

Dados $a,b,k\in\mathbb{Z}$ con $b,k\neq 0$, es fácil ver que $\displaystyle\frac{k\cdot a}{k\cdot b}:=\frac{a}{b}$. Esto nos permite escribir cada número racional de manera unívoca como $\displaystyle\frac{a}{b}$ donde $a\in\mathbb{Z}$, $b\in\mathbb{N}$ y $\text{mcd}\pr{a,b}=1$.

\begin{proposition}
\rm Definimos las operaciones $+:\mathbb{Q}\times\mathbb{Q}\rightarrow\mathbb{Q}$ y $\cdot:\mathbb{Q}\times\mathbb{Q}\rightarrow\mathbb{Q}$ como sigue:

\begin{equation}
    \frac{a}{b}+\frac{c}{d}:=\frac{a\cdot d+b\cdot c}{b\cdot d},\hspace{0.5cm}\frac{a}{b}\cdot\frac{c}{d}:=\frac{a\cdot b}{c\cdot d}.
\end{equation}

Se cumplen las siguientes propiedades:

\begin{itemize}
    \item[1.] $+$ y $\cdot$ están bien definidas, es decir, la suma y el producto de números racionales no dependen de los representantes escogidos.
    \item[2.] $\pr{\mathbb{Q},+,\cdot}$ es un cuerpo, es decir, verifica las siguientes propiedades:
    \begin{itemize}
      \item Asociatividad: Para todo $p,q,r\in\mathbb{Q}$, $p+\pr{q+r}=\pr{p+q}+r$ y $p\cdot\pr{q\cdot r}=\pr{p\cdot q}\cdot r$.
      \item Conmutatividad: Para todo $p,q\in\mathbb{Q}$, $p+q=q+p$ y $p\cdot q=q\cdot p$.
      \item Existencia y unicidad de elemento neutro de la suma: Existe $e\in\mathbb{Q}$ tal que $x+e=e+x=x$ para todo $x\in\mathbb{Q}$. Además, $e$ es único y lo denotaremos como $0$.
      \item Existencia y unicidad del elemento opuesto de un número: Dado $x\in\mathbb{Q}$, existe $y=y\pr{x}\in\mathbb{Q}$ tal que $x+y=y+x=0$. Además, para cada $x\in\mathbb{Q}$, $y$ es único y lo denotaremos como $-x$. Dados $x,y\in\mathbb{Q}$, con $y\neq 0$, denotaremos por $\displaystyle\frac{x}{y}:=x\cdot y^{-1}$.
      \item Existencia y unicidad de elemento neutro del producto: Existe $e'\in\mathbb{Q}$ tal que $x\cdot e=e\cdot x=x$ para todo $x\in\mathbb{Q}$. Además, $e$ es único y lo denotaremos como $1$.
      \item Existencia y unicidad del elemento inverso de un número: Dado $x\in\mathbb{Q}\setminus\set{0}$, existe $y=y\pr{x}\in\mathbb{Q}$ tal que $x\cdot y=y\cdot x=1$. Además, para cada $x\in\mathbb{Q}$, $y$ es único y lo denotaremos como $x^{-1}$. 
      \item Distributividad del producto respecto de la suma: Para todo $p,q,r\in\mathbb{Q}$, $p\cdot \pr{q+r}=p\cdot q+p\cdot r$.
    \end{itemize}
    \item[3.] Regla de los signos: Para todo $x,y\in\mathbb{Q}$, $\pr{-1}\cdot x=-x$, $-\pr{-x}=x$, $\pr{-x}\cdot y=x\cdot\pr{-y}=-\pr{x\cdot y}$ y $\pr{-x}\cdot\pr{-y}=x\cdot y$.
    \item[4.] Para todo $x,y\in\mathbb{Q}\setminus\set{0}$, $\frac{1}{x}=x^{-1}$, $\pr{x^{-1}}^{-1}=x$ y $\pr{x\cdot y}^{-1}=x^{-1}\cdot y^{-1}$.
    \item[5.] Ley Cancelativa de la Suma: Para todo $x,y,z\in\mathbb{Q}$, si $x+z=y+z$, entonces $x=y$.
    \item[6.] Ley Cancelativa del Producto: Para todo $x,y,z\in\mathbb{Q}$ con $z\neq 0$, $x\cdot z=y\cdot z$, entonces $x=y$.
    \item[7.] Para todo $x\in\mathbb{Q}$, $x\cdot 0=0\cdot x=0$. Más aún, $\pr{\mathbb{Q},+,\cdot}$ es un dominio de integridad, es decir, si $x,y\in\mathbb{Q}$ son tales que $x\cdot y=0$, entonces $x=0$ o $y=0$.
\end{itemize}
\end{proposition}

\begin{definition}
  \rm Dados $x:=\frac{a}{b},y:=\frac{c}{d}\in\mathbb{Q}$ dos números racionales con $b,d\in\mathbb{N}$, diremos que:

  \begin{itemize}
    \item \textit{$x$ es menor que $y$}, y lo denotaremos como $x<y$, si $a\cdot d<b\cdot c$. Diremos que \textit{$x$ es menor o igual que $y$}, y lo denotaremos como $x\leq y$, si $x<y$ o $x=y$.
    \item \textit{$x$ es mayor que $y$}, y lo denotaremos como $x>y$, si $y<x$. Diremos que \textit{$x$ es mayor o igual que $y$}, y lo denotaremos como $x\geq y$, si $y\leq x$.
    \item \textit{$x$ es positivo} (resp. \textit{$x$ es no negativo}) si $x>0$ (resp. $x\geq 0$).
    \item \textit{$x$ es negativo} (resp. \textit{$x$ es no positivo}) si $x<0$ (resp. $x\leq 0$).
  \end{itemize}
\end{definition}

\begin{proposition}
  \rm La relación $\leq$ sobre $\mathbb{Q}$ verifica las siguientes propiedades:
  \begin{itemize}
    \item[1.] $\leq$ está bien definida, es decir, no depende de los representantes escogidos.
    \item[2.] $\leq$ es una relación de orden, es decir, $\leq$ es reflexiva, antisimétrica y transitiva.
    \item[3.] Para todo $x,y\in\mathbb{Q}$, si $x<y$, entonces $-y<-x$. En particular, si $x>0$, entonces $-x<0$.
    \item[4.] Para todo $x\in\mathbb{Q}$, si $x>0$ (resp. $x<0$), entonces $x^{-1}>0$ (resp. $x^{-1}<0$).
    \item[5.] Para todo $x,y\in\mathbb{Q}$, si $x<y$, entonces $y^{-1}<x^{-1}$ 
    \item[6.] Ley de Tricotomía: Dados $x,y\in\mathbb{Z}$, se tiene una y solamente una de las siguientes relaciones: $x<y$, $x=y$ o $x>y$. En consecuencia, $\leq$ es una relación de orden total.
    \item[7.] Compatibilidad del orden con la suma de números enteros: Dados $x,y,z\in\mathbb{Q}$, se tiene que $x<y$ si y solamente si $x+z<y+z$.
    \item[8.] Compatibilidad del orden con el producto de números enteros: Dados $x,y,z\in\mathbb{Q}$ con $z>0$ (resp. $z<0$), se tiene que $x<y$ si y solamente si $x\cdot z<y\cdot z$ (resp. $x\cdot z>y\cdot z$).
  \end{itemize}
\end{proposition}



\begin{proposition}
  \rm Sea $f:\mathbb{Z}\rightarrow\mathbb{Q}$ dada por $\displaystyle f\pr{n}=\frac{n}{1}$. Se cumplen las siguientes propiedades:

  \begin{itemize}
    \item[1.] $f$ es inyectiva.
    \item[2.] Para todo $n,m\in\mathbb{Z}$, se cumple que $f\pr{n+m}=f\pr{n}+f\pr{m}$ y $f\pr{n\cdot m}=f\pr{n}\cdot f\pr{m}$.
    \item[3.] Para todo $n,m\in\mathbb{Z}$, $n<m$ si y solamente si $f\pr{n}<f\pr{m}$. 
  \end{itemize}
  \begin{proof}[\rm\textbf{Demostración}]
    \hfill
    \begin{itemize}
    \item[1.] Dados $n,m\in\mathbb{Z}$, si $f\pr{n}=f\pr{m}$, entonces $\frac{n}{1}=\frac{m}{1}$, es decir, $n=n\cdot 1=1\cdot m=m$. Por tanto, $f$ es inyectiva.
    \item[2.] En efecto, se tiene lo siguiente:
    
    \begin{equation}
        \begin{split}
            f\pr{n+m} & =\frac{n+m}{1}=\frac{n\cdot 1+1\cdot m}{1\cdot 1}=\frac{n}{1}+\frac{m}{1}=f\pr{n}+f\pr{m},\\
            \\
            f\pr{n\cdot m} & =\frac{n\cdot m}{1}=\frac{n\cdot m}{1\cdot 1}=\frac{n}{1}\cdot\frac{m}{1}=f\pr{n}\cdot f\pr{m}.
        \end{split}
    \end{equation}
    \item[3.] $f\pr{n}<f\pr{m}$ si y solamente si $\frac{n}{1}<\frac{m}{1}$, esto es, $n=n\cdot 1<1\cdot m=m$.
  \end{itemize}
  \end{proof}
  
\end{proposition}

La proposición anterior nos permite ver $\mathbb{Z}$ como subconjunto de $\mathbb{Q}$. Dado $n\in\mathbb{Z}$, denotaremos por $n:=\frac{n}{1}$. 

\begin{proposition}
  \rm $\mathbb{Q}$ es un conjunto infinito numerable.
  \begin{proof}[\rm\textbf{Demostración}]
    Como $\mathbb{N}\subseteq\mathbb{Z}\subseteq\mathbb{Q}$, necesariamente, $\mathbb{Q}$ es un conjunto infinito. Sea $\psi:\mathbb{Q}\rightarrow\mathbb{N}$ dada por $\psi\pr{\frac{a}{b}}=2\cdot 3^a\cdot 5^b$ si $a\geq 0$ y $\psi\pr{\frac{a}{b}}=3^{-a}\cdot 5^b$ si $a<0$ donde $a\in\mathbb{Z}$, $b\in\mathbb{N}$ y $\text{mcd}\pr{a,b}=1$. Obsérvese que $\psi$ está bien definida y es inyectiva por el \hyperref[Teorema Fundamental de la Aritmética]{Teorema Fundamental de la Aritmética}. 
    
    Sea $S:=\set{2^k\cdot 3^a\cdot 5^b\middle|k\in\set{0,1},a\in\mathbb{N}_0,b\in\mathbb{N},\text{mcd}\pr{a,b}=1}\subseteq\mathbb{N}$. Basta observar que $\mathbb{Q}\cong S$ y $S$ es infinito numerable.
    
  \end{proof}
\end{proposition}

\begin{proposition}
    \rm $\mathbb{Q}$ es el menor cuerpo conteniendo a $\mathbb{Z}$, es decir, si $\pr{\mathbb{K},+,\cdot}$ es un cuerpo y $f:\mathbb{Z}\rightarrow\mathbb{K}$ es un homomorfismo inyectivo de anillos, existe un homomorfismo inyectivo de cuerpos $g:\mathbb{Q}\rightarrow\mathbb{K}$.
    \begin{proof}[\rm\textbf{Demostración}]
        Sea $f:\mathbb{Z}\rightarrow\mathbb{K}$ un homomorfismo inyectivo de anillos. Definimos $g:\mathbb{Q}\rightarrow\mathbb{K}$ como $g\pr{\frac{a}{b}}=\frac{f\pr{a}}{f\pr{b}}$ donde $a\in\mathbb{Z}$ y $b\in\mathbb{N}$. Como $f$ es inyectivo, se tiene que $f\pr{b}\neq 0_{\mathbb{K}}$ para todo $n\in\mathbb{N}$ donde $0_{\mathbb{K}}$ es el elemento neutro de la suma en $\mathbb{K}$. Veamos que $g$ está bien definida, es decir, el valor de $g$ no depende del representante escogido. Sean $a,a'\in\mathbb{Z}$ y $b,b'\in\mathbb{N}$ tales que $\frac{a}{b}=\frac{a'}{b'}$, es decir, $a\cdot b'=a'\cdot b$. Entonces $f\pr{a}\cdot f\pr{b'}=f\pr{a\cdot b'}=f\pr{a'\cdot b}=f\pr{a'}\cdot f\pr{b}$. Esto prueba que $\frac{f\pr{a}}{f\pr{b}}=\frac{f\pr{a'}}{f\pr{b'}}$.

        Por otro lado, dados $a,c\in\mathbb{Z}$ y $b,d\in\mathbb{N}$, se tiene lo siguiente:

        \begin{equation}
            \begin{split}
                g\pr{\frac{a}{b}}+g\pr{\frac{c}{d}} & =\frac{f\pr{a}}{f\pr{b}}+\frac{f\pr{c}}{f\pr{d}}=\frac{f\pr{a}\cdot f\pr{d}+f\pr{b}\cdot f\pr{c}}{f\pr{b}\cdot f\pr{d}}=\frac{f\pr{a\cdot d+b\cdot c}}{f\pr{b\cdot d}}\\
                & =g\pr{\frac{a\cdot d+b\cdot c}{b\cdot d}}=g\pr{\frac{a}{b}+\frac{c}{d}},\\
                g\pr{\frac{a}{b}}\cdot g\pr{\frac{c}{d}} & =\frac{f\pr{a}}{f\pr{b}}\cdot\frac{f\pr{c}}{f\pr{d}}=\frac{f\pr{a}\cdot f\pr{c}}{f\pr{b}\cdot f\pr{d}}=\frac{f\pr{a\cdot c}}{f\pr{b\cdot d}}\\
                & =g\pr{\frac{a\cdot b}{c\cdot d}}=g\pr{\frac{a}{b}\cdot\frac{c}{d}}.
            \end{split}
        \end{equation}

        Esto prueba que $g$ es un homorfismo. Además, si $g\pr{\frac{a}{b}}=0$, entonces $f\pr{a}=0$ lo cual solo es posible si $a=0$ pues $f$ es inyectiva. Por tanto, $g$ es inyectiva.
    \end{proof}
\end{proposition}

\begin{proposition}[\textbf{$\mathbb{Q}$ es denso}]
    \rm Dados $r,s\in\mathbb{Q}$ con $r<s$, existe $t\in\mathbb{Q}$ tal que $r<t<s$. 
    \begin{proof}[\rm\textbf{Demostración}]
        Basta tomar $t:=\frac{1}{2}\cdot\pr{r+s}$ y observar que $t>\frac{1}{2}\cdot\pr{r+r}=\frac{1}{2}\cdot\pr{2\cdot r}=r$ y $t<\frac{1}{2}\cdot\pr{s+s}=\frac{1}{2}\cdot\pr{2\cdot s}=s$.
    \end{proof}
\end{proposition}

\begin{proposition}[\rm\textbf{$\mathbb{Q}$ es arquimediano}]
    \rm Dado $r\in\mathbb{Q}$ con $r>0$, existe $n=n\pr{r}\in\mathbb{N}$ tal que $\frac{1}{n}<r$.
    \begin{proof}[\rm\textbf{Demostración}]
        Sean $a,b\in\mathbb{N}$ tales que $r=\frac{a}{b}$. Tenemos que encontrar $n\in\mathbb{N}$ tal que $\frac{1}{n}<\frac{a}{b}$, o, equivalentemente, $b<n\cdot a$. Por reducción al absurdo, supongamos que $b\geq n\cdot a$ para todo $n\in\mathbb{N}$. Sea $S:=\set{b-n\cdot a\middle|n\in\mathbb{N}}\subseteq\mathbb{N}_0$. Como $S$ es no vacío, por el Principio de Buena Ordenación, existe $n_0\in\mathbb{N}$ tal que $\min S:=b-n_0\cdot a$. Sin embargo, $b-\pr{n_0+1}\cdot a\in S$ y $b-\pr{n_0+1}\cdot a\cdot a<b-n_0\cdot a$ lo cual es absurdo. 
    \end{proof}
\end{proposition}

\section{Incompletitud de \texorpdfstring{$\mathbb{Q}$}{Q}}

Todo lo dicho hasta aquí habla de ``lo que funciona bien'' en $\mathbb{Q}$. Destacamos ahora algunas de sus carencias fundamentales que se resumen diciendo que $\mathbb{Q}$ es incompleto.

\textbf{1. Existencia de longitudes no expresables mediante números racionales.} Consideremos un cuadrado de lado $1$ al que queremos calcular la longitud de su diagonal. Por el Teorema de Pitágoras, debe cumplirse la siguiente ecuación:

\begin{equation}
  x^2=1^2+1^2=2.
\end{equation}

Veamos que esta ecuación no tiene solución en $\mathbb{Q}$. Supongamos que existe $x\in\mathbb{Q}$ tal que $x^2=2$. Entonces existen $a\in\mathbb{Z}$ y $b\in\mathbb{N}$ con $\text{mcd}\pr{a,b}=1$ tales que $x=\frac{a}{b}$. Despejando, tenemos que $a^2=2\cdot b^2$, es decir, $a^2$ es par. Como todo número par al cuadrado es par y todo número impar al cuadrado es impar, se concluye que $a$ es par. Luego existe $k\in\mathbb{Z}$ tal que $a=2\cdot k$. Sustituyendo, obtenemos que $4\cdot k^2=2\cdot b^2$ y, simplificando, llegamos a que $b^2=2\cdot k^2$, esto es, $b^2$ es par y, en consecuencia, $b$ también. Hemos llegado a una contradicción pues $\text{mcd}\pr{a,b}=1$. Por tanto, $x^2=2$ no tiene solución en $\mathbb{Q}$.

Se puede demostrar de manera análoga que la ecuación $x^2=p$ con $p\in\mathbb{N}$ primo no tiene solución.

\begin{definition}[\rm\textbf{Conjunto acotado}]
  \rm Sea $\pr{X,\leq}$ un conjunto parcialmente ordenado. Dado $A\subseteq X$, diremos que:

  \begin{itemize}
    \item $A$ está acotado superiormente si existe $x\in X$ tal que $a\leq x$ para todo $a\in A$. En este caso, diremos que $x$ es una cota superior de $A$.
    \item $A$ está acotado inferiormente si existe $x\in X$ tal que $x\leq a$ para todo $a\in A$. En este caso, diremos que $x$ es una cota inferior de $A$.
    \item $A$ está acotado si $A$ está acotado superior e inferiormente.
    \item $\alpha\in X$ es el supremo de $A$ si $A$ es no vacío y está acotado superiormente, $\alpha$ es una cota superior de $A$ y $\alpha\leq x$ para toda cota superior $x$ de $A$. En este caso, denotaremos $\alpha=\sup A$. Si $\alpha\in A$, diremos que $\alpha$ es el máximo de $A$ y lo denotaremos como $\alpha=\max A$.
    \item $\alpha\in X$ es el ínfimo de $A$ si $A$ es no vacío y está acotado inferiormente, $\alpha$ es una cota inferior de $A$ y $x\leq \alpha$ para toda cota superior $x$ de $A$. En este caso, denotaremos $\alpha=\inf A$. Si $\alpha\in A$, diremos que $\alpha$ es el mínimo de $A$ y lo denotaremos como $\alpha=\min A$.
  \end{itemize}
\end{definition}


\begin{definition}[\rm\textbf{Conjunto ordenado completo}]
  \rm Un conjunto parcialmente ordenado $\pr{X,\leq}$ se dice completo si todo subconjunto no vacío de $X$ acotado superiormente tiene supremo.
\end{definition}

\textbf{2. Existen conjuntos de números racionales no vacíos y acotados superiormente que no tienen supremo.} Consideremos el conjunto $S:=\set{r\in\mathbb{Q}\middle|r^2<2}$. Es fácil ver que $S$ es no vacío. Además, $S$ está acotado superiormente por $2$ pues si $r\in\mathbb{Q}$ con $r\geq 2$, entonces $r^2\geq 4>2$, es decir, $r\notin S$. Por reducción al absurdo, supongamos que existe $\alpha\in\mathbb{Q}$ tal que $\alpha=\sup S$. Necesariamente, $\alpha^2\neq 2$. Tenemos dos casos:

\begin{itemize}
  \item Supongamos que $\alpha^2<2$. Sea $h\in\mathbb{Q}$ positivo cualquiera. Entonces:
  
  \begin{equation}
    \pr{\alpha+h}^2=\alpha^2+2\cdot\alpha\cdot h+h^2=\alpha^2+\pr{2\cdot\alpha+h}\cdot h.
  \end{equation}

  Por la densidad de $\mathbb{Q}$, tomamos $\displaystyle h<\min\set{1,\frac{2-\alpha^2}{2\cdot\alpha+1}}$. Entonces:

  \begin{equation}
    \pr{\alpha+h}^2<\alpha^2+\pr{2\cdot\alpha+1}\cdot h<\alpha^2+\pr{2-\alpha^2}=2,
  \end{equation}

  es decir, $\alpha+h\in S$. Esto es absurdo pues $\alpha$ es el supremo de $S$ y $\alpha+h>\alpha$.

  \item Supongamos que $\alpha^2>2$. Sea $h\in\mathbb{Q}$ positivo cualquiera. Entonces:
  
  \begin{equation}
    \pr{\alpha-h}^2=\alpha^2-2\cdot\alpha\cdot h+h^2.
  \end{equation}

  Por la densidad de $\mathbb{Q}$, tomamos $h$ tal que $\alpha^2-2\cdot\alpha\cdot h+h^2>2+h^2$, es decir, $\displaystyle h<\frac{\alpha^2-2}{2\cdot\alpha}$. Entonces:

  
  \begin{equation}
    \pr{\alpha-h}^2>\alpha^2+\pr{h-2\cdot\alpha}\cdot h>2+h^2>2,
  \end{equation}

  esto es, $\alpha-h\in S$. Esto es absurdo pues $\alpha$ es el supremo de $S$ y $\alpha-h<\alpha$.
\end{itemize}

En conclusión, $S$ no tiene supremo en $\mathbb{Q}$. Por tanto, $\pr{\mathbb{Q},\leq}$ no es un conjunto parcialmente ordenado completo.


\begin{definition}[\rm\textbf{Valor absoluto}]
  \rm Dado $r\in\mathbb{Q}$, definimos el valor absoluto de $r$ como el siguiente valor:

  \begin{equation}
    \bars{r}:=\begin{cases}
      r & \text{si $r\geq 0$}\\
      -r & \text{si $r<0$}
    \end{cases}
  \end{equation}
\end{definition}

\begin{definition}[\textbf{Sucesiones de números racionales}]
  \rm Una sucesión de números racionales es una función $\varphi:\mathbb{N}\rightarrow\mathbb{Q}$. Denotaremos $\varphi:=\pr{r_n}_{n\in\mathbb{N}}$ donde $r_n:=\varphi\pr{n}$ para todo $n\in\mathbb{N}$. Dada una sucesión de números racionales $\pr{r_n}_{n\in\mathbb{N}}$, diremos que:

  \begin{itemize}
    \item $\pr{r_n}_{n\in\mathbb{N}}$ es creciente (resp. decreciente) si $r_n\leq r_{n+1}$ (resp. $r_n\geq r_{n+1}$) para todo $n\in\mathbb{N}$. % chktex 12
    \item $\pr{r_n}_{n\in\mathbb{N}}$ está acotada superiormente (resp. inferiormente) si existe $k\in\mathbb{Q}$ tal que $r_n\leq k$ (resp. $r_n\geq k$) para todo $n\in\mathbb{N}$. % chktex 12
    \item $\pr{r_n}_{n\in\mathbb{N}}$ es convergente en $\mathbb{Q}$ y su límite es $r\in\mathbb{Q}$ si para todo $\varepsilon\in\mathbb{Q}$ con $\varepsilon>0$, existe $n_0=n_0\pr{\varepsilon}\in\mathbb{N}$ tal que, para todo $n\in\mathbb{N}$ con $n\geq n_0$, se cumple que $\bars{r_n-r}<\varepsilon$.
  \end{itemize}
\end{definition}

Se puede probar fácilmente que si $\pr{r_n}_{n\in\mathbb{N}}$ y $\pr{s_n}_{n\in\mathbb{N}}$ son dos sucesiones de números racionales convergentes a $r,s\in\mathbb{Q}$ respectivamente, entonces $\pr{r_n+s_n}_{n\in\mathbb{N}}$ y $\pr{r_n\cdot s_n}_{n\in\mathbb{N}}$ son convergentes a $r+s$ y $r\cdot s$ respectivamente. Además, si $s_n\neq 0$ para todo $n\in\mathbb{N}$ y $s\neq 0$, entonces $\displaystyle\pr{\frac{r_n}{s_n}}_{n\in\mathbb{N}}$ es convergente a $\frac{r}{s}$.

\textbf{3. Existen sucesiones crecientes en $\mathbb{Q}$ acotadas superiormente que no tienen límite racional. } Consideremos la siguiente expresión:

\begin{equation}
  r=1+\frac{1}{1+\frac{2}{1+\frac{1}{1+\frac{2}{\dots}}}}
\end{equation}

Observemos que $r$ verifica la siguiente ecuación:

\begin{equation}
  r=1+\frac{1}{1+\dfrac{2}{r}}.
\end{equation}

Sea $\pr{r_n}_{n\in\mathbb{N}}$ la sucesión de números racionales definida por recurrencia como $r_1:=1$ y:

\begin{equation}
  r_{n+1}:=1+\frac{1}{1+\dfrac{2}{r_n}}=1+\frac{r_n}{r_n+2}=\frac{2+2\cdot r_n}{2+r_n},
\end{equation}

para todo $n\in\mathbb{N}$. Obsérvese que $\pr{r_n}_{n\in\mathbb{N}}$ está bien definida, es decir, $r_n\in\mathbb{Q}$ para todo $n\in\mathbb{N}$ pues, por inducción, es fácil ver que $r_n>0$ para todo $n\in\mathbb{N}$. Veamos que $r_n<2$ para todo $n\in\mathbb{N}$. En efecto, para $n=1$, se tiene que $r_1=1<2$. Supongamos que $r_n<2$ para $n\in\mathbb{N}$. Entonces:

\begin{equation}
  r_{n+1}=\frac{2+2\cdot r_n}{2+r_n}=1+\frac{r_n}{2+r_n}<1+\frac{2}{2}=1+1=2.
\end{equation}

Por el Principio de Inducción, se tiene que $r_n<2$ para todo $n\in\mathbb{N}$. Veamos que $\pr{r_n}_{n\in\mathbb{N}}$ es creciente. En efecto, para $n=1$, se tiene que $r_1=1<\frac{4}{3}=r_2$. Supongamos que $r_n<r_{n+1}$ para $n\in\mathbb{N}$. Entonces:

\begin{equation}
  \begin{split}
    r_{n+2}-r_{n+1} & =\frac{2+2\cdot r_{n+1}}{2+r_{n+1}}-\frac{2+2\cdot r_n}{2+r_n}=\frac{\pr{2+2\cdot r_{n+1}}\cdot\pr{2+r_n}-\pr{2+r_{n+1}}\cdot\pr{2+2\cdot r_n}}{\pr{2+r_{n+1}}\cdot\pr{2+r_n}}\\
    &=\frac{\pr{4+2\cdot r_n+4\cdot r_{n+1}+2\cdot r_n\cdot r_{n+1}}-\pr{4+4\cdot r_n+2\cdot r_{n+1}+2\cdot r_n\cdot r_{n+1}}}{\pr{2+r_{n+1}}\cdot\pr{2+r_n}}\\
    &=\frac{2\cdot \pr{r_{n+1}-r_n}}{\pr{2+r_{n+1}}\cdot\pr{2+r_n}}>0.
  \end{split}
\end{equation}

Por el Principio de Inducción, se tiene que $\pr{r_n}_{n\in\mathbb{N}}$ es creciente. Por reducción al absurdo, supongamos que $\pr{r_n}_{n\in\mathbb{N}}$ converge a $r\in\mathbb{Q}$. Entonces:

\begin{equation}
  r=\frac{2+2\cdot r}{2+r}\Rightarrow r\cdot\pr{2+r}=2+2\cdot r\Rightarrow r^2+2\cdot r=2\cdot r+2\Rightarrow r^2=2.
\end{equation}

Esto es una contradicción. Por tanto, $\pr{r_n}_{n\in\mathbb{N}}$ es una sucesión de números racionales creciente y acotada superiormente que no tiene límite racional.

\section{Fracciones y números decimales}

\begin{definition}
  \rm Sean $r\in\mathbb{Q}$, $c\in\mathbb{Z}$, $b\in\mathbb{N}$ con $b>1$ y $\pr{a_n}_{n\in\mathbb{N}}$ una sucesión de dígitos en $\set{0,1,\dots,b-1}$. Diremos que $r$ tiene en base $b$ la expresión decimal $\pr{c,a_1a_2\dots a_n\dots}_b$ si la sucesión de números racionales $\pr{s_n}_{n\in\mathbb{N}}$ dada por:

  \begin{equation}
    s_n:=c+\sum_{k=1}^{n}{\frac{a_k}{b^k}}
  \end{equation}

  converge a $r$. En este caso, se dice que:

  \begin{itemize}
    \item la expresión decimal $r=\pr{c,a_1a_2\dots a_n\dots}_b$ es exacta si existe $n_0\in\mathbb{N}$ tal que $a_n=0$ para todo $n\in\mathbb{N}$ con $n>n_0$. En tal caso, escribiremos $r=c,a_1a_2\dots a_{n_0}$.
    \item la expresión decimal $r=\pr{c,a_1a_2\dots a_n\dots}_b$ es periódica si existen $n_0\in\mathbb{N}_0$ y $p\in\mathbb{N}$ tal que $a_{n+p}=a_n$ para todo $n\in\mathbb{N}$ con $n>n_0$. En tal caso, escribiremos $r:=\pr{c,a_1a_2\dots a_{n_0}\overline{a_{n_0+1}\dots a_p}}_b$. Si $n_0=0$, diremos que la expresión decimal anterior es periódica pura y, si $n_0>0$, diremos que la expresión decimal anterior es periódica mixta.
  \end{itemize}
\end{definition}

\begin{theorem}
  \rm Todo número racional tiene una expresión decimal que será exacta o periódica. Además, si $\displaystyle r=\frac{m}{n}\in\mathbb{Q}$ con $m\in\mathbb{Z}$, $n\in\mathbb{N}$ y $\text{mcd}\pr{m,n}=1$, y $b\in\mathbb{N}$ con $b>1$, se cumplen las siguientes propiedades:

  \begin{itemize}
    \item $r$ tiene expresión decimal exacta en base $b$ si y solamente si, para todo $p\in\mathbb{N}$ primo, si $p\mid n$, entonces $p\mid b$. 
    \item $r$ tiene expresión decimal periódica en base $b$ si y solamente si existe $p\in\mathbb{N}$ primo tal que $p\mid n$ y $p\nmid b$.
  \end{itemize}
\end{theorem}

A partir de la expresión decimal (exacta o periódica) de un número racional, podemos obtener su expresión como fracción (generatriz) de la siguiente forma:

\begin{itemize}
  \item  Si la expresión decimal de $r$ es exacta, es decir, $r=\pr{c,a_1a_2\dots a_{n_0}}_b$, entonces $b^{n_0}\cdot r=\pr{ca_1a_2\dots a_{n_0}}_b$ y, por tanto:
  
  \begin{equation}
    r=\frac{\pr{ca_1a_2\dots a_{n_0}}_b}{b^{n_0}}.
  \end{equation}
  \item  Si la expresión decimal de $r$ es periódica, es decir, $r=\pr{c,a_1a_2\dots a_{n_0}\overline{a_{n_0+1}\dots a_p}}_b$, entonces $\pr{b^{n_0+p}-b^{n_0}}\cdot r=\pr{ca_1a_2\dots a_{n_0}a_{n_0+1}\dots a_{n_0+p}}_b-\pr{ca_1a_2\dots a_{n_0}}_b$ y, por tanto:
  
  \begin{equation}
    r=\frac{\pr{ca_1a_2\dots a_{n_0}a_{n_0+1}\dots a_{n_0+p}}_b-\pr{ca_1a_2\dots a_{n_0}}_b}{b^{n_0+p}-b^{n_0}}.
  \end{equation}
\end{itemize}
