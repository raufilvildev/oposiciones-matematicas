\chapter{Conceptos básicos de la teoría de conjuntos. Estructuras algebraicas.}

\section{Teoría de conjuntos}

El matemático alemán Georg Cantor, creador de la Teoría de Conjuntos, introdujo el concepto de conjunto como \textit{una colección bien definida de entes que reciben el nombre de elementos del conjunto}. 

Esta forma de introducir los conjuntos no es plenamente satisfactoria pues conduce a contradicciones como la conocida como \textit{paradoja de Russel}. 

La Teoría de Conjuntos moderno no da ninguna definición operativa de conjunto y, en su lugar, postula la existencia de los conjuntos mediante una colección de axiomas conocidos como los \textit{Axiomas de Zermelo-Fraenkel}. Mencionaremos algunos de ellos que son los que necesitaremos para nuestro propósito.

\subsection{Igualdad de conjuntos. Conjunto vacío. Subconjuntos}

\textbf{Axioma de extensionalidad}. Dos conjuntos $X$ e $Y$ son iguales si y solamente si tienen los mismos elementos:

\begin{equation}
  \pr{\forall X}\pr{\forall Y}\pr{\forall z}\pr{\pr{z\in X\iff z\in Y}\implies X=Y}.
\end{equation}

La implicación contraria es consecuencia directa del axioma de igualdad en lógica de primer orden.

\textbf{Axioma del conjunto vacío.} Existe un conjunto que no tiene elementos:

\begin{equation}
  \pr{\exists X}\pr{\forall Y}\pr{Y\notin X}.
\end{equation}


\begin{proposition}
  \rm Existe un único conjunto que no tiene elementos. Este conjunto lo llamaremos conjunto vacío y será denotado por $\varnothing$.
\end{proposition}


\begin{definition}[\textbf{Subconjunto}]
\rm Sean $X$ y $Y$ dos conjuntos. Diremos que $X$ es un subonjunto de $Y$ o que $X$ está contenido en $Y$, y lo escribiremos como $X\subseteq Y$, si todo elemento de $X$ también es un elemento de $Y$:

\begin{equation}
  X\subseteq Y\triangleq \forall z\pr{z\in X\implies z\in Y}
\end{equation}
\end{definition}

\begin{proposition}
\rm Sean $X$ e $Y$ conjuntos. Se cumplen las siguientes propiedades:
\begin{itemize}
  \item[1.] $X=Y$ si y solamente si $X\subseteq Y$ e $Y\subseteq X$.
  \item[2.] $\varnothing\subseteq X$ y $X\subseteq X$. 
\end{itemize}
\end{proposition}

\subsection{Pares ordenados}

\textbf{Axioma de pares.} Dados dos conjuntos $X$ e $Y$, existe un conjunto $Z$ cuyos únicos elementos son $X$ e $Y$:

\begin{equation}
  \pr{\forall X}\pr{\forall Y}\pr{\exists Z}\pr{\forall z}\pr{z\in Z\iff z=X\lor z=Y}.
\end{equation}

\begin{proposition}
  \rm Dados dos conjuntos $X$ e $Y$, existe un único conjunto $Z$ cuyos elementos son $X$ e $Y$ que denotaremos por $\set{X,Y}$. Además, $\set{X,Y}=\set{Y,X}$.
\end{proposition}


Dado un conjunto $X$, denotaremos por $\set{X}:=\set{X,X}$.

\begin{definition}[\textbf{Pares ordenados}]
  \rm Dados dos conjuntos $X$ e $Y$, definimos el par ordenado $\pr{X,Y}$ como el siguiente conjunto:

  \begin{equation}
    \pr{X,Y}:=\set{X,\set{X,Y}}.
  \end{equation}

  Inductivamente, si $X_1,\dots,X_n, X_{n+1}$ son $n+1$ conjuntos, definimos el $\pr{n+1}$-upla $\pr{X_1,\dots,X_n,X_{n+1}}$ como el siguiente conjunto:

  \begin{equation}
    \pr{X_1,\dots,X_n,X_{n+1}}:=\pr{\pr{X_1,\dots,X_n},X_{n+1}}.
  \end{equation}
\end{definition}


\begin{proposition}
  \rm Sean $A$, $B$, $C$ y $D$ cuatro conjuntos. Entonces $\pr{A,B}=\pr{C,D}$ si y solamente si $A=C$ y $B=D$. Inductivamente, si $X_1,\dots,X_n,Y_1,\dots,Y_n$ son $2\cdot n$ conjuntos, se tiene que $\pr{X_1,\dots,X_n}=\pr{Y_1,\dots,Y_n}$ si y solamente si $X_k=Y_k$ para todo $k\in\set{1,\dots,n}$.
\end{proposition}

De la proposición anterior se deduce que si $A$, $B$ son dos conjuntos, entonces $\pr{A,B}\neq\pr{B,A}$ generalmente.

\subsection{Unión, intersección, diferencia y producto cartesiano}

\textbf{Axioma de la unión.} Para todo conjunto $X$, existe un conjunto $Y$ cuyos elementos son los elementos de los elementos de $X$:

\begin{equation}
  \pr{\forall X}\pr{\exists Y}\pr{\forall z}\pr{z\in Y\iff \pr{\exists Z}\pr{Z\in X\land z\in Z}}.
\end{equation}

\begin{proposition}
  \rm Dado un conjunto $\mathcal{X}$, existe un único conjunto cuyos elementos son los elementos de $X$. Este conjunto lo llamaremos conjunto unión y será denotado por $\displaystyle\bigcup{\mathcal{X}}:=\set{z\middle|\pr{\exists X}\pr{X\in\mathcal{X}\land z\in X}}$. Otra notación empleada es $\displaystyle\bigcup_{X\in\mathcal{X}}{X}$.
\end{proposition}

Inductivamente, si $X_1,\dots,X_n,X_{n+1}$ son $n+1$ conjuntos, definimos el conjunto cuyos únicos elementos son $X_1,\dots,X_n,X_{n+1}$ como sigue:

\begin{equation}
  \set{X_1,\dots,X_n,X_{n+1}}:=\bigcup{\set{\set{X_1,\dots,X_n},\set{X_{n+1}}}}.
\end{equation}

Por otro lado, si $X_1,\dots,X_n$ son $n$ conjuntos, definimos su unión como el siguiente conjunto:

\begin{equation}
  \bigcup_{k=1}^{n}{X_k}:=\bigcup{\set{X_1,\dots,X_n}}.
\end{equation}

Otra notación empleada es $X_1\cup\dots\cup X_n$.

\begin{proposition}
  \rm Sean $X$, $Y$ y $Z$ tres conjuntos. Se cumple:

  \begin{itemize}
    \item[1.] Dado un conjunto $z$, se tiene que $z\in X\cup Y$ si y solamente si $z\in X$ o $z\in Y$:
    
    \begin{equation}
      X\cup Y:=\set{z\middle|z\in X\lor z\in Y}.
    \end{equation}
    \item[2.] Conmutatividad: $X\cup Y=Y\cup X$.
    \item[3.] Asociatividad: $\pr{X\cup Y}\cup Z=X\cup\pr{Y\cup Z}$.
    \item[4.] Elemento neutro: $X\cup \varnothing=\varnothing\cup X=X$.
    \item[5.] $X,Y\subseteq X\cup Y$.
    \item[6.] $X\subseteq Y$ si y solamente si $X\cup Y=Y$.
\end{itemize}
\end{proposition}

\textbf{Axioma de las partes.} Dado un conjunto $X$, existe un conjunto $Y$ cuyos elementos son todos los suconjuntos de $X$:

\begin{equation}
  \pr{\forall X}\pr{\exists Y}\pr{\forall Z}\pr{Z\in Y\iff Z\subseteq X}.
\end{equation}

\begin{proposition}
  \rm Dado un conjunto $X$, existe un único conjunto cuyos elementos son todos los conjuntos de $X$. Este conjunto lo llamaremos conjunto de las partes de $X$ y lo denotaremos por $\mathcal{P}\pr{X}:=\set{A\middle|A\subseteq X}$.
\end{proposition}

\textbf{Esquema axiomático de separación}. Sea $\varphi\pr{u,p_1, \dots,p_n}$ una fórmula del lenguaje de la Teoría de Conjuntos. Para todo conjunto $X$, $p_1$, \dots, $p_n$, existe un conjunto $Y$ cuyos elementos son aquellos conjuntos $x$ tales que $x\in X$ y verifican $\varphi\pr{x,p_1,\dots,p_n}$ para todo los conjuntos $p_1,\dots,p_n$:

\begin{equation}
    \pr{\forall X}\pr{\forall p_1}\dots\pr{\forall p_n}\pr{\exists Y}\pr{\forall z}\pr{z\in Y\iff z\in X\land \varphi\pr{z,p_1,\dots,p_n}}
\end{equation}

\begin{proposition}
  \rm Sean $\varphi\pr{u,p_1, \dots,p_n}$ una fórmula del lenguaje de la Teoría de Conjuntos y $X$ un conjunto. Entonces existe un único conjunto cuyos elementos son aquellos $x$ tales que $x\in X$ y verifican $\varphi\pr{x,p_1,\dots,p_n}$ para todo los conjuntos $p_1,\dots,p_n$. 
  
  Dados $n$ conjuntos $p_1,\dots,p_n$, el conjunto anterior lo denotaremos como $\set{x\in X\middle|\varphi\pr{x,p_1,\dots,p_n}}\subseteq X$. Es posible ver el conjunto anterior denotado como $\set{x\middle|\varphi\pr{x,p_1,\dots,p_n}}$ cuando se deduce que $x\in X$ de la fórmula $\varphi\pr{x,p_1,\dots,p_n}$.
\end{proposition}


\begin{definition}[\textbf{Intersección de conjuntos}]
  \rm Sea $\mathcal{X}$ un conjunto no vacío. Se define la intersección de $\mathcal{X}$ como el conjunto formado por aquellos elementos de elementos de $X$ que están en todos los elementos de $X$:

  \begin{equation}
    \bigcap{\mathcal{X}}:=\set{x\middle|\pr{\forall Z}\pr{Z\in\mathcal{X}\implies x\in Z}}.
  \end{equation}

  Otra notación empleada es $\displaystyle\bigcap_{X\in\mathcal{X}}{X}$.

  Sean $X_1,\dots,X_n$ $n$ conjuntos. Se define la intersección de $X_1,\dots,X_n$ como el conjunto:

  \begin{equation}
    \bigcap_{k=1}^{n}{X_k}:=\bigcap{\set{X_1,\dots,X_n}}.
  \end{equation}

  Otra notación empleada es $X_1\cap\dots\cap X_n$.

  Dados dos conjuntos $X$ e $Y$, diremos que son disjuntos si $X\cap Y=\varnothing$.
\end{definition}

\begin{proposition}
\rm Sean $X$, $Y$ y $Z$ tres conjuntos. Se cumple:

\begin{itemize}
    \item[1.] Dado un conjunto $z$, se tiene que $z\in X\cap Y$ si y solamente si $z\in X$ y $z\in Y$:
    
    \begin{equation}
      X\cap Y:=\set{z\middle|z\in X\land z\in Y}.
    \end{equation}
    \item[2.] Conmutatividad: $X\cap Y=Y\cap A$.
    \item[3.] Asociatividad: $\pr{X\cap Y}\cap Z=X\cap\pr{Y\cap Z}$.
    \item[4.] $X\cap Y\subseteq X,Y$.
    \item[5.] $X\subseteq Y$ si y solamente si $X\cap Y=X$.
    \item[6.] Elemento disipativo: $X\cap\varnothing=\varnothing\cap X=\varnothing$.
\end{itemize}
\end{proposition}

\begin{theorem}[\textbf{Leyes distributivas}]
\rm Sean $X$ e $\mathcal{Y}$ dos conjuntos. Se cumple:

\begin{itemize}
    \item[1.] $\displaystyle X\cap\pr{\bigcup{\mathcal{Y}}}=\bigcup_{Y\in\mathcal{Y}}{\pr{X\cap Y}}=\bigcup{\set{X\cap Y\middle|Y\in\mathcal{Y}}}$. En particular, si $A$, $B$ y $C$ son tres conjuntos, tenemos que:
    
    \begin{equation}
        A\cap\pr{B\cup C}=\pr{A\cap B}\cup\pr{A\cap C}.
    \end{equation}
    \item[2.] $\displaystyle X\cup\pr{\bigcap{\mathcal{Y}}}=\bigcap_{Y\in\mathcal{Y}}{\pr{X\cup Y}}=\bigcap{\set{X\cup Y\middle|Y\in\mathcal{Y}}}$. En particular, si $A$, $B$ y $C$ son tres conjuntos, tenemos que:

    \begin{equation}
        A\cup\pr{B\cap C}=\pr{A\cup B}\cap\pr{A\cup C}.
    \end{equation}
\end{itemize}
\end{theorem}

\begin{definition}[\textbf{Diferencia de conjuntos}]
  \rm Sean $X$ y $Y$ dos conjuntos. Definimos la diferencia entre $X$ e $Y$ como el conjunto:

\begin{equation}
    X\setminus Y:=\set{x\middle|x\in X\land x\notin Y}.
\end{equation}
\end{definition}

\begin{definition}[\textbf{Complementario de un conjunto}]
\rm Fijado un conjunto universal $U$, es decir, un conjunto que contiene a todos los posibles conjuntos del problema que tratamos, definimos el complementario de un conjunto $X$, y lo denotaremos como $\overline{X}$ o $X^c$, como el conjunto $U\setminus A$.
\end{definition}

\begin{theorem}[\textbf{Leyes de De Morgan}]
  \rm Sean $X$ e $\mathcal{Y}$ dos conjuntos. Se cumple:

  \begin{itemize}
      \item[1.] $X\setminus\pr{\bigcup{\mathcal{Y}}}=\bigcap_{Y\in\mathcal{Y}}{\pr{X\setminus Y}}=\bigcap{\set{X\setminus Y\middle|Y\in\mathcal{Y}}}$. En particular, si $A$, $B$ y $C$ son tres conjuntos, tenemos que:
      
      \begin{equation}
          A\setminus\pr{B\cup C}=\pr{A\setminus B}\cap\pr{A\setminus C}.
      \end{equation}
      \item[2.] $X\setminus\pr{\bigcap{\mathcal{Y}}}=\bigcup_{Y\in\mathcal{Y}}{\pr{X\setminus Y}}=\bigcup{\set{X\setminus Y\middle|Y\in\mathcal{Y}}}$. En particular, si $A$, $B$ y $C$ son tres conjuntos, tenemos que:

      \begin{equation}
          A\setminus\pr{B\cap C}=\pr{A\setminus B}\cup\pr{A\setminus C}.
      \end{equation}
  \end{itemize}
\end{theorem}

\begin{definition}[\textbf{Producto cartesiano}]
  \rm Sean $X_1,\dots,X_n$ $n$ conjuntos. Se define el producto cartesiano de $X_1,\dots,X_n$ como el conjunto:

  \begin{equation}
    \prod_{k=1}^{n}{X_k}:=\set{\pr{x_1,\dots,x_n}\middle|x_1\in X_1\land\dots\land x_n\in X_n}.
  \end{equation}

  Otra notación empleada es $X_1\times\cdots\times X_n$. Si $X$ es un conjunto tal que $X=X_1=\cdots=X_n$, denotaremos por $X^n:=\overbrace{X\times\cdots\times X}^{n\text{ veces}}$.
\end{definition}

\subsection{Aplicaciones}

\begin{definition}[\textbf{Aplicación}]
  \rm Sean $X$ e $Y$ dos conjuntos. Una aplicación (de $X$ en $Y$) es una terna $f=\pr{X,Y,G}$ tal que $G\subseteq X\times Y$ verifica que si $x\in X$ e $y,y'\in Y$ tales que $\pr{x,y},\pr{x,y'}\in G$, entonces $y=y'$:

\begin{equation}
    \pr{\forall x}\pr{\forall y}\pr{\forall y'}\pr{\pr{x,y}\in G\land\pr{x,y'}\in G\rightarrow y=y'}.
\end{equation}

Sean $x\in X$ e $y\in Y$. Si $\pr{x,y}\in G$, escribiremos $f\pr{x}=y$ y diremos que $y$ es el valor de $x$ por $f$. Por otro lado, si $f$ es una aplicación de $X$ en $Y$, escribiremos $f:X\rightarrow Y$. 

En este contexto, diremos que:

\begin{itemize}
    \item $X$ es el dominio de $f$ y lo escribiremos como $\text{dom}\pr{f}:=X$.
    \item $Y$ es el codominio de $f$ y lo escribiremos como $\text{cod}\pr{f}:=Y$.
    \item $\text{im}\pr{f}:=\set{y\in Y\middle|\pr{\exists x}\pr{x\in X\rightarrow f\pr{x}=y}}$ es la imagen de $f$. 
    \item $G$ es el grafo de $f$ y lo escribiremos como $\text{graf}\pr{f}:=G$.
\end{itemize}
\end{definition}

\begin{definition}[\textbf{Restricción de una aplicación}]
  \rm Sean $f:X\rightarrow Y$ una aplicación y $U\subseteq X$. Se define la restricción de $f$ a $U$ como la aplicación $\left.f\right|_U:U\rightarrow Y$ dada por $\left.f\right|_U\pr{x}=f\pr{x}$.
\end{definition}


\begin{definition}[\textbf{Inyectividad, sobreyectividad y biyectividad}]
  \rm Sean $X$ e $Y$ dos conjuntos y $f:X\rightarrow Y$ una aplicación. Diremos que:

\begin{itemize}
    \item $f$ es inyectiva si para todo $x,x'\in X$, $f\pr{x}=f\pr{x'}$ implica que $x=x'$. Para indicarlo, escribiremos $f:X\hookrightarrow Y$.
    \item $f$ es sobreyectiva si para todo $y\in Y$, existe $x\in X$ tal que $f\pr{x}=y$. Para indicarlo, escribiremos $f:X\twoheadrightarrow Y$.
    \item $f$ es biyectiva si es inyectiva y sobreyectiva. Para indicarlo, escribiremos $f:X\xrightarrow{\cong}Y$.
\end{itemize}
\end{definition}

\begin{definition}[\textbf{Imagen directa e inversa}]
\rm Sean $f:X\rightarrow Y$ una aplicación, $A\subseteq X$ y $B\subseteq Y$. Definimos:

\begin{itemize}
    \item la imagen directa de $A$ como el conjunto:

    \begin{equation}
        f\pr{A}:=\set{y\in Y\middle|\pr{\exists x}\pr{x\in X\land f\pr{x}=y}}=\set{f\pr{x}\middle|x\in X}.
    \end{equation}

    Obsérvese que $\text{im}\pr{f}=f\pr{X}$.
    \item la imagen inversa de $B$ como el conjunto:

    \begin{equation}
        f^{-1}\pr{B}:=\set{x\in X\middle|f\pr{x}\in B}.
    \end{equation}

    
    Obsérvese que $X=f^{-1}\pr{Y}$.
\end{itemize}
\end{definition}

\begin{proposition}
\rm Sea $f:X\rightarrow Y$ una aplicación. Se cumple:

\begin{itemize}
    \item[1.] Si $A_1, A_2\subseteq X$ y $B_1,B_2\subseteq Y$ tales que $A_1\subseteq A_2$ $B_1\subseteq B_2$, entonces $f\pr{A_1}\subseteq f\pr{A_2}$ y $f^{-1}\pr{B_1}\subseteq f^{-1}\pr{B_2}$.
    \item[2.] Si $\mathcal{X}\subseteq\mathcal{P}\pr{X}$ y $\mathcal{Y}\subseteq\mathcal{P}\pr{Y}$, entonces:
    
    \begin{equation}
        f\pr{\bigcup{\mathcal{X}}}=\bigcup_{A'\in\mathcal{X}}{f\pr{A'}},\hspace{0.5cm}  f^{-1}\pr{\bigcup{\mathcal{Y}}}=\bigcup_{B'\in\mathcal{Y}}{f^{-1}\pr{B'}}.
    \end{equation}
    \item[3.] Si $\mathcal{X}\subseteq\mathcal{P}\pr{X}$ y $\mathcal{Y}\subseteq\mathcal{P}\pr{Y}$, entonces:
    \begin{equation}
        f\pr{\bigcap{\mathcal{X}}}\subseteq\bigcap_{A'\in\mathcal{X}}{f\pr{A'}}, \hspace{0.5cm} f^{-1}\pr{\bigcap{\mathcal{Y}}}=\bigcap_{B'\in\mathcal{B}}{f^{-1}\pr{B'}}. 
    \end{equation}
  \end{itemize}
\end{proposition}
\begin{definition}[\textbf{Composición de aplicaciones}]
  \rm Sean $f:X\rightarrow Y$ y $g:Y\rightarrow Z$ dos aplicaciones. Se define la composición de $f$ con $g$ como la aplicación $g\circ f:X\rightarrow Z$ dada por $\pr{g\circ f}\pr{x}=g\pr{f\pr{x}}$.
\end{definition}

\begin{definition}[\textbf{Aplicación invertible}]
  \rm Sea $f:X\rightarrow Y$ una aplicación. Diremos que $f$ es invertible si existe una aplicación $g:Y\rightarrow X$, que llamaremos aplicación inversa de $f$, tal que $g\circ f=\text{id}_X$ y $f\circ g=\text{id}_Y$.
\end{definition}

\begin{proposition}
  \rm La aplicación inversa de una aplicación invertible es única.
\end{proposition}

\begin{proposition}
  \rm Sea $f:X\rightarrow Y$ una aplicación. Se cumple que $f$ es invertible si y solamente si $f$ es biyectiva.
\end{proposition}

\subsection{Relaciones}
\begin{definition}[\rm\textbf{Relación}]

\rm Una relación $n$-aria sobre un conjunto $X$ es un subconjunto $R\subseteq X^n$. Comúnmente, escribiremos $R\pr{x_1,\dots,x_n}$ para indicar que $\pr{x_1,\dots,x_n}\in R$.

Si $n=2$, diremos que $R$ es una relación binaria en $X$ y escribiremos $xRy$, que leeremos como \textit{$x$ está relacionado con $y$}, para indicar que $\pr{x,y}\in R$. Cuando la relación sea obvia por el contexto, escribiremos $x\thicksim y$ en lugar de $xRy$. En este caso, diremos que:

\begin{itemize}
  \item $R$ es reflexiva si $xRx$ para todo $x\in X$.
  \item $R$ es simétrica si para todo $x,y\in X$, se tiene que $xRy$ implica que $yRx$.
  \item $R$ es antisimétrica si para todo $x,y\in X$, se tiene que $xRy$ e $yRx$ implica que $x=y$.
  \item $R$ es transitiva si para todo $x,y,z\in X$, se tiene que $xRy$ e $yRz$ implica que $xRz$.
\end{itemize}
\end{definition}

\begin{definition}[\rm\textbf{Relación de equivalencia}]
  \rm Sean $X$ un conjunto y $R$ una relación (binaria) sobre $X$. Diremos que $R$ es una relación de equivalencia si $\thicksim$ es reflexiva, simétrica y transitiva.
\end{definition}

\begin{definition}[\textbf{Clase de equivalencia}]
\rm Sean $X$ un conjunto, $x\in X$ y $R$ una relación de equivalencia sobre $X$. Se define la clase de equivalencia de $x$ como el conjunto $R\pr{x}:=\set{y\in X\middle|xRy}$. Cuando la relación sea obvia por el contexto, escribiremos $\br{x}$ o $\overline{x}$ en lugar de $R\pr{x}$. Los elementos de $R\pr{x}$ se denominan representantes de la clase.
\end{definition}

\begin{definition}[\textbf{Conjunto cociente}]
\rm Sean $X$ un conjunto y $R$ una relación de equivalencia sobre $X$. Definimos el conjunto cociente como sigue:

\begin{equation}
    X/R=\set{R\pr{x}\middle|x\in X}.
\end{equation}

\end{definition}

\section{Estructuras algebraicas}

\begin{definition}[\textbf{Operación interna}]
\rm Dado un conjunto no vacío $X$, una operación interna sobre $X$ es una aplicación $\ast: X\times X\rightarrow X$. Dados $x,y\in X$, denotaremos por $x\ast y:=\ast\pr{x,y}$. Si $\ast$ es una operación interna sobre $X$, diremos que:

\begin{itemize}
  \item $\ast$ es asociativa si para todo $x,y,z\in X$, se cumple que $x\ast\pr{y\ast z}=\pr{x\ast y}\ast z$.
  \item $\ast$ es conmutativa si para todo $x,y\in X$, se cumple que $x\ast y=y\ast x$.
  \item $e\in X$ es el elemento neutro de $\ast$ si para todo $x\in X$, se cumple que $e\ast x=x\ast e=x$. 
  \item $x\in X$ es invertible respecto de $\ast$ si $\ast$ tiene elemento neutro y existe $x'\in X$ que cumple que $x\ast x'=x'\ast x=e$.
  \item $\ast$ es distributiva respecto a otra operación interna $\star$ sobre $X$ si para todo $x,y,z\in X$, se cumple que $x\ast\pr{y\star z}=\pr{x\ast y}\star\pr{x\ast z}$.
  \item $x\in X$ es simplificable respecto a $\ast$ si para todo $y,z\in X$, se cumple que $x\ast y=x\ast z$ implica $y=z$.
  \item $\ast$ es compatible respecto a una relación de equivalencia $\thicksim$ sobre $X$ si para todo $x,x',y,y'\in X$, se cumple que $x\thicksim x'$ e $y\thicksim y'$ implica que $x\ast y\thicksim x'\ast y'$. En tal caso, $\ast$ induce una operación interna sobre el conjunto cociente $X/\thicksim$, denotada con el mismo símbolo $\ast$, llamada operación cociente y que se define como $\br{x}\ast\br{y}:=\br{x\ast y}$ para $\br{x},\br{y}\in X/\thicksim$.
\end{itemize}
\end{definition}

\begin{proposition}
  \rm Sean $X$ un conjunto no vacío y $\ast$ una operación interna sobre $X$. Se cumplen las siguientes propiedades:

  \begin{itemize}
    \item Si $\ast$ tiene elemento neutro, entonces es único.
    \item Si $\ast$ tiene elemento neutro, entonces este es invertible y simplificable respecto de $\ast$.
    \item Si $\ast$ es asociativa y tiene elemento neutro, y $x\in X$ es invertible respecto de $\ast$, entonces el elemento inverso de $x$ es único y lo denotaremos como $x^{-1}$. Además, $\pr{x^{-1}}^{-1}=x$.
    \item Si $\ast$ es asociativa y tiene elemento neutro, y $x,y\in X$ son invertibles respecto de $\ast$, entonces $x\ast y$ es invertible respecto de $\ast$ y se cumple que $\pr{x\ast y}^{-1}=y^{-1}\ast x^{-1}$. 
    \item Si $\ast$ es asociativa y tiene elemento neutro, dados $a,b\in X$ con $a$ invertible respecto de $\ast$, la ecuación $a\ast x=b$ tiene solución única $x=a^{-1}\ast b$ y la ecuación $y\ast a=b$ tiene solución única $y=b\ast a^{-1}$.
    \item Si $\ast$ es asociativa y tiene elemento neutro, y $x\in X$ es invertible respecto de $\ast$, entonces $x$ es simplificable respecto de $\ast$.
  \end{itemize}
\end{proposition}

\begin{definition}[\textbf{Estructura algebraica}]
  \rm Sean $X$ un conjunto no vacío y $\ast_1,\dots,\ast_n$ $n$ operaciones internas sobre $X$. Diremos que la $\pr{n+1}$-upla $\pr{X,\ast_1,\dots,\ast_n}$ es una estructura algebraica.
\end{definition}

A continuación, trataremos algunas estructuras algebraicas de interés: grupos, anillos y cuerpos.

\subsection{Grupos}

\begin{definition}[\textbf{Grupo}]
  \rm Sean $G$ un conjunto no vacío y $\ast$ una operación interna sobre $X$. Diremos que la pareja $\pr{G,\ast}$ es un grupo si se cumplen las siguientes condiciones:

  \begin{itemize}
    \item[1.] $\ast$ es asociativa.
    \item[2.] $\ast$ tiene elemento neutro.
    \item[3.] Para todo $x\in G$, existe un elemento inverso de $x$ respecto a $\ast$.
  \end{itemize}

  Si además $\ast$ es conmutativa, diremos que $\pr{G,\ast}$ es un grupo abeliano.
\end{definition}

\begin{definition}[\textbf{Subgrupo}]
  \rm Sean $\pr{G,\ast}$ un grupo y $H\subseteq G$ no vacío. Diremos que $H$ es un subgrupo de $G$ (respecto de $\ast$) si se cumplen las siguientes propiedades:

  \begin{itemize}
    \item[1.] Para todo $x,y\in H$, se cumple que $x\ast y\in H$.
    \item[2.] Si $e\in G$ es el elemento neutro de $\ast$, entonces $e\in H$.
    \item[3.] Para todo $x\in H$, se cumple que $x^{-1}\in H$.
  \end{itemize}

  En particular, $\pr{H,\ast}:=\pr{H,\left.\ast\right|_{H\times H}}$ es un grupo.
\end{definition}

\begin{theorem}[\rm\textbf{Caracterización de subgrupos}]
  \rm Sean $\pr{G,\ast}$ un grupo y $H\subseteq G$ no vacío. Entonces $H$ es un subgrupo de $G$ (respecto de $\ast$) si y solo si para todo $x,y\in H$, se cumple que $x\ast y^{-1}\in H$.
\end{theorem}

\begin{definition}[\textbf{Homomorfismo de grupos}]
  \rm Sean $\pr{G,\ast}$ y $\pr{H,\star}$ dos grupos y $f:G\rightarrow H$ una aplicación. Diremos que $f$ es un homomorfismo (del grupo $\pr{G,\ast}$ en el grupo $\pr{H,\star}$) si para todo $x,y\in G$, se cumple que $f\pr{x\ast y}=f\pr{x}\star f\pr{y}$. Si $f:G\rightarrow H$ es un homomorfismo, diremos que:

  \begin{itemize}
    \item $f$ es un monomorfismo si $f$ es inyectiva.
    \item $f$ es un epimorfismo si $f$ es sobreyectiva.
    \item $f$ es un isomorfismo si $f$ es biyectiva. En tal caso, diremos que los grupos $\pr{G,\ast}$ y $\pr{H,\star}$ son isomorfos y lo denotaremos como $\pr{G,\ast}\cong\pr{H,\star}$.
    \item $f$ es un endomorfismo si $\pr{G,\ast}=\pr{H,\star}$.
    \item $f$ es un automorfismo si $f$ es un endomorfismo e isomorfismo.
  \end{itemize}
\end{definition}

\begin{proposition}
  \rm Sean $\pr{G,\ast}$ y $\pr{H,\star}$ dos grupos con elementos neutros $e_G\in G$ y $e_H\in H$ respectivamente, y $f:G\rightarrow H$ un homomorfismo. Se cumplen las siguientes propiedades:
  \begin{itemize}
    \item[1.] $f\pr{e_G}=e_H$.
    \item[2.] Para todo $x\in G$, se cumple que $f\pr{x^{-1}}=f\pr{x}^{-1}$.
    \item[3.] Si $S$ es un subgrupo de $G$, entonces $f\pr{S}$ es un subgrupo de $H$.
    \item[4.] Si $T$ es un subgrupo de $H$, entonces $f^{-1}\pr{T}$ es un subgrupo de $G$.
    \item[5.] Si $\pr{J, \odot}$ es un grupo y $g:H\rightarrow J$ un homomorfismo, entonces $g\circ f:G\rightarrow J$ es un homomorfismo. 
  \end{itemize}
\end{proposition}

\begin{definition}[\textbf{Núcleo de un homomorfismo de grupos}]
  \rm Sean $\pr{G,\ast}$ y $\pr{H,\star}$ dos grupos con elementos neutros $e_G\in G$ y $e_H\in H$ respectivamente, y $f:G\rightarrow H$ un homomorfismo. Definimos el núcleo de $f$ como el conjunto $\ker f:=\set{x\in G\middle|f\pr{x}=e_H}=f^{-1}\pr{e_H}$.
\end{definition}

\begin{proposition}
  \rm Sean $\pr{G,\ast}$ y $\pr{H,\star}$ dos grupos y $f:G\rightarrow H$ un homomorfismo. Se cumplen las siguientes propiedades:
  \begin{itemize}
    \item[1.] $\ker f$ es un subgrupo de $G$.
    \item[2.] $f$ es inyectiva si y solamente si $\ker f=\set{e_G}$.
    \item[3.] $\text{im} f$ es un subgrupo de $H$.
  \end{itemize}
\end{proposition}
\subsection{Anillos}

\begin{definition}[\textbf{Anillo}]
  \rm Sean $R$ un conjunto y $+,\cdot:R\times R\rightarrow R$ dos operaciones internas sobre $R$. Diremos que la terna $\pr{R,+,\cdot}$ es un anillo si $\pr{R,+}$ es un grupo abeliano y $\cdot$ es asociativa, distributiva respecto de $+$ y tiene elemento neutro. 
  
  Al elemento neutro de $+$ lo denotaremos como $0$ y al de $\cdot$ como $1$. Diremos que $\pr{R,+,\cdot}$ es un anillo no trivial si $0\neq 1$.
  
  Dado $r\in R$, su elemento inverso respecto a $+$ lo denotaremos por $-r$ y diremos que $-r$ es el opuesto de $r$. Dados $r,s\in R$, denotaremos por $r-s:=r+\pr{-s}$.
  
  Si $r\in R$ es invertible respecto al producto, diremos que $r$ es una unidad (respecto a $\pr{R,+,\cdot}$). El conjunto de unidades en $R$ lo denotaremos por $R^*$.

  Si $\cdot$ es conmutativo, diremos que $\pr{R,+,\cdot}$ es un anillo conmutativo. 
\end{definition}

\begin{proposition}
  \rm Sea $\pr{R,+,\cdot}$ un anillo. Se cumplen las siguientes propiedades:

  \begin{itemize}
    \item Elemento disipativo: Para todo $r\in R$, $r\cdot 0=0\cdot r=0$.
    \item Regla de los signos: Para todo $r,s\in R$, se cumple que $r\cdot\pr{-s}=\pr{-r}\cdot s=-\pr{r\cdot s}$ y $\pr{-r}\cdot\pr{-s}=r\cdot s$.
    \item Si $\pr{R,+,\cdot}$ es un anillo no trivial, entonces $0$ no es una unidad.
    \item $\pr{R^*,\cdot}$ es un grupo.
  \end{itemize}
\end{proposition}

\begin{definition}[\textbf{Divisor de cero en un anillo}]
  \rm Sea $\pr{R,+,\cdot}$ un anillo. Diremos que $r\in R\setminus\set{0}$ es un divisor de cero si existe $s\in R\setminus\set{0}$ tal que $r\cdot s=0$. Si $R$ no posee divisores de $0$, diremos que $\pr{R,+,\cdot}$ es un dominio de integridad.
\end{definition}

\begin{definition}[\textbf{Característica de un anillo}]
  \rm Sea $\pr{R,+.\cdot}$ un anillo con $0_R$ como elemento neutro de $+$ y $1_R$ como elemento neutro de $\cdot$, $r\in R$ y $n\in\mathbb{N}$. Definimos:

  \begin{equation}
    n\cdot r:=\overbrace{r+\cdots+r}^{n\text{ veces}}.
  \end{equation}

  Diremos que $R$ tiene característica finita si existe $n\in\mathbb{N}$ tal que $n\cdot 1_R=0_R$. Si $R$ tiene característica finita, denotaremos por $\text{car}\pr{R}$ al menor $n\in\mathbb{N}$ que verifica que $n\cdot 1_R=0_R$ y lo llamaremos característica de $R$. Si $R$ no tiene característica finita, diremos que $R$ tiene característica infinita.
\end{definition}

\begin{proposition}
  \rm Sea $\pr{R,+,\cdot}$ un anillo. Se tiene que $r\in R\setminus\set{0}$ es un elemento simplificable si y solamente si $r$ no es divisor de $0$.
\end{proposition}

\begin{definition}[\textbf{Subanillo}]
  \rm Sean $\pr{R,+,\cdot}$ un anillo y $S\subseteq R$. Diremos que $S$ es un subanillo de $R$ si se cumplen las siguientes propiedades:
  
  \begin{itemize}
    \item[1.] $S$ es un subgrupo de $\pr{R,+}$.
    \item[2.] $1\in S$.
    \item[3.] Para todo $s,t\in S$, $s\cdot t\in S$.
  \end{itemize} 
\end{definition}

\begin{theorem}[\textbf{Caracterización de subanillos}]
  \rm Sean $\pr{R,+,\cdot}$ un anillo y $S\subseteq R$ no vacío. Se tiene que $S$ es un subanillo de $R$ si y solamente si para todo $s,t\in S$, $s-t,s\cdot t\in S$.
\end{theorem}

\begin{definition}[\textbf{Ideal de un anillo}]
  \rm Sean $\pr{R,+,\cdot}$ un anillo e $I\subseteq R$. Diremos que $I$ es un ideal de $R$ si $\pr{I,+}$ es un subgrupo de $R$ respecto a $+$ y, para todo $r\in R$ y $s\in I$, $r\cdot s\in I$.
\end{definition}

\begin{definition}[\textbf{Homomorfismo de anillos}]
  \rm Sean $\pr{R,+,\cdot}$ y $\pr{S,\oplus,\odot}$ dos anillos con elementos neutros para el producto $1_R$ y $1_S$ respectivamente, y $f:R\rightarrow S$ una aplicación. Diremos que $f$ es un homomorfismo (del anillo $\pr{R,+,\cdot}$ al anillo $\pr{S,\oplus,\odot}$) si se cumplen las siguientes propiedades:

  \begin{itemize}
    \item[1.] $f$ es un homomorfismo del grupo $\pr{R,+}$ al grupo $\pr{S,\oplus}$.
    \item[2.] Para todo $r,s\in R$, $f\pr{r\cdot s}=f\pr{r}\odot f\pr{s}$.
    \item[3.] $f\pr{1_R}=1_S$.
  \end{itemize}

  Si $f:R\rightarrow S$ es un homomorfismo, diremos que:

  \begin{itemize}
    \item $f$ es un monomorfismo si $f$ es inyectiva.
    \item $f$ es un epimorfismo si $f$ es sobreyectiva.
    \item $f$ es un isomorfismo si $f$ es biyectiva. En tal caso, diremos que los anillos $\pr{R,+,\cdot}$ y $\pr{S,\oplus,\odot}$ son isomorfos y lo denotaremos como $\pr{R,+,\cdot}\cong\pr{S,\oplus,\odot}$.
    \item $f$ es un endomorfismo si $\pr{R,+,\cdot}=\pr{S,\oplus,\odot}$.
    \item $f$ es un automorfismo si $f$ es un endomorfismo e isomorfismo.
  \end{itemize}

  De manera análoga a como hicimos para grupos, podemos definir monomorfismo, epimorfismo, isomorfismo, endomorfismo y automorfismo.
\end{definition}

\begin{proposition}
  \rm Sean $\pr{R,+,\cdot}$ y $\pr{S,\oplus,\odot}$ dos anillos y $f:R\rightarrow S$ un homomorfismo. Se cumplen las siguientes propiedades:

  \begin{itemize}
    \item[1.] Para todo $r\in R$, $f\pr{-r}=-f\pr{r}$.
    \item[2.] Si $R'$ es un subanillo de $R$, entonces $f\pr{R'}$ es un subgrupo de $S$.
    \item[3.] Si $S'$ es un subgrupo de $S$, entonces $f^{-1}\pr{S'}$ es un subgrupo de $R$.
    \item[4.] Si $\pr{T, \boxplus , \boxdot}$ es un anillo y $g:S\rightarrow T$ un homomorfismo, entonces $g\circ f:R\rightarrow T$ es un homomorfismo. 
  \end{itemize}
\end{proposition}

\begin{definition}[\textbf{Núcleo de un homomorfismo de anillos}]
  \rm Sean $\pr{R,+,\cdot}$ y $\pr{S,\oplus,\odot}$ dos anillos con elementos neutros para la suma $0_R$ y $0_S$ respectivamente, y $f:R\rightarrow S$ un homomorfismo. Definimos el núcleo de $f$ como el conjunto $\ker f:=\set{x\in R\middle|f\pr{x}=0_S}=f^{-1}\pr{0_S}$.
\end{definition}

\begin{proposition}
  \rm Sean $\pr{R,+,\cdot}$ y $\pr{S,\oplus,\odot}$ dos anillos y $f:R\rightarrow S$ un homomorfismo. Se cumplen las siguientes propiedades:

  \begin{itemize}
    \item[1.] $\ker f$ es un subanillo de $R$.
    \item[2.] $f$ es inyectiva si y solamente si $\ker f=\set{1_R}$, donde $1_R$ es el elemento neutro de $+$.
    \item[3.] $\text{im} f$ es un subanillo de $S$.  
  \end{itemize}
\end{proposition}
\subsection{Cuerpos}

\begin{definition}[\textbf{Cuerpo}]
  \rm Se dice que un anillo $\pr{\mathbb{K},+,\cdot}$ es un cuerpo si es un anillo conmutativo tal que $\mathbb{K}^*=\mathbb{K}\setminus\set{0}$. 
\end{definition}

\begin{definition}[\textbf{Subcuerpo}]
  \rm Sea $\pr{\mathbb{K},+,\cdot}$ un cuerpo y $H\subseteq\mathbb{K}$. Diremos que $H$ es un subcuerpo de $\mathbb{K}$ si $H$ es un subanillo de $\mathbb{K}$ y $\pr{H,+,\cdot}$ es un cuerpo.
\end{definition}

\begin{proposition}
  \rm Sean $\pr{\mathbb{K},+,\cdot}$ y $\pr{\mathbb{H},\oplus, \odot}$ dos cuerpos y $f:\mathbb{K}\rightarrow\mathbb{H}$ una aplicación. Entonces $f$ es un homomorfismo si y solamente si para todo $x,y\in\mathbb{K}$, $f\pr{x+y}=f\pr{x}\oplus f\pr{y}$ y $f\pr{x\cdot y}=f\pr{x}\odot f\pr{y}$.
\end{proposition}

\begin{proposition}
  \rm Sean $\pr{\mathbb{K},+,\cdot}$ y $\pr{\mathbb{H},\oplus, \odot}$ dos cuerpos y $f:\mathbb{K}\rightarrow\mathbb{H}$ un homomorfismo. Se cumplen las siguientes propiedades:

  \begin{itemize}
    \item[1.] $\ker f=\set{0_\mathbb{K}}$ o $\ker f=\mathbb{K}$.
    \item[2.] $f$ es inyectiva si y solamente si $f\not\equiv 0$.
    \item[3.] $\text{im} f$ es un subcuerpo de $\mathbb{H}$.
  \end{itemize}
\end{proposition}
